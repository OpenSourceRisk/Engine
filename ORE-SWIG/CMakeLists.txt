find_package(SWIG REQUIRED)
find_package(Python REQUIRED COMPONENTS Development)

# Load the UseSWIG module from the cmake installation
include(UseSWIG)

# Path to the top level .i file.
set(OREA_SWIG_FILE ${CMAKE_CURRENT_SOURCE_DIR}/OREAnalytics-SWIG/SWIG/oreanalytics.i)

# Tell SWIG to compile in C++ mode
set_property(SOURCE ${OREA_SWIG_FILE} PROPERTY CPLUSPLUS ON)

# This was only added in 3.20. Should we use it given that we support 3.15?
# https://cmake.org/cmake/help/latest/module/UseSWIG.html
# Added in MR https://gitlab.acadiasoft.net/qs/oreplus/-/merge_requests/3204
set_property(SOURCE ${OREA_SWIG_FILE} PROPERTY USE_SWIG_DEPENDENCIES TRUE)

# Tell SWIG to build a Python module, to be called "OREP"
# We sometimes put a hyphen in the variable (e.g. "-gd").
# Function swig_add_library() (defined in module UseSWIG.cmake) misappropriates this variable in two ways,
# and in both cases, a hyphen causes an error:
# 1) swig_add_library() uses the variable as part of the name of a function in the autogenerated C source code,
#    and a hyphen causes a compilation error.
# 2) swig_add_library() uses the variable as part of the package name,
#    and a hyphen results in the package not getting loaded at runtime.
# So unset the variable before calling swig_add_library(),
# then restore it after, because it is required by subsequent calls to target_link_libraries().

# Need to do the unset & restore all CMAKE_<CONFIG>_POSTFIX after update.
set(configs DEBUG RELEASE RELWITHDEBINFO MINSIZEREL)

# Store original and unset.
foreach(cfg IN LISTS configs)
    set(var_name "CMAKE_${cfg}_POSTFIX")
    set(orig_postfix_${cfg} "${${var_name}}")
    unset(${var_name})
endforeach()

# Run the command.
swig_add_library(OREP TYPE MODULE LANGUAGE python SOURCES ${OREA_SWIG_FILE})

# Restore.
foreach(cfg IN LISTS configs)
    set(var_name "CMAKE_${cfg}_POSTFIX")
    set(${var_name} "${orig_postfix_${cfg}}")
endforeach()

# Add to the list of SWIG include directories
foreach(swig_dir OREAnalytics OREData QuantExt QuantLib)
    set_property(TARGET OREP APPEND PROPERTY
        SWIG_INCLUDE_DIRECTORIES ${CMAKE_CURRENT_SOURCE_DIR}/${swig_dir}-SWIG/SWIG
    )
endforeach()

set_property(TARGET OREP APPEND PROPERTY SWIG_COMPILE_OPTIONS -fastdispatch -includeall -small)

# Add all libraries to link with
target_link_libraries(OREP PRIVATE ${OREA_LIB_NAME} Boost::iostreams)

if(ORE_USE_ZLIB)
    target_link_libraries(OREP PRIVATE ZLIB::ZLIB)
    target_compile_definitions(OREP PRIVATE ORE_USE_ZLIB)
endif()

# On windows, do not tell the build which python libs to use.  It automatically 
# looks for python release libs, even in debug, and invoking 
# target_link_libraries(OREP PRIVATE Python::Python) breaks the debug build.
if(MSVC)
    target_include_directories(OREP PRIVATE ${Python_INCLUDE_DIRS})
    target_link_directories(OREP PRIVATE ${Python_LIBRARY_DIRS})
    remove_compile_options(OREP "/we4189")
else()
    target_link_libraries(OREP PRIVATE Python::Python)
    # Need to remove those, clang throws errors with the autogenerated swig c++ wrapper code
    remove_compile_options(OREP "-Werror=unused-variable" "-Werror=uninitialized")
endif()
